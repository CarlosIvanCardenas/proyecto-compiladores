program factorial_recursive

fun factorial(n: int): int {
    var val: int
    var i: int
    if (n < 0) {
        write("Incorrect input")
        val = 0
    } else {
        val = 1
        for i = 1 to n+1 {
            val = val * i
        }
    }
    return (val)
}

main() {
    var n: int
    var f: int
    write("Escribe valor de n: ")
    read(n)
    f = factorial(n)
    write(f)
}
